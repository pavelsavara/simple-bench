@code {
    List<string> _steps = new() { "First" ,  "Second" };
    int _index;

    private void Add()
    {
        Random rnd = new Random();
        _steps.Add( $"Step {rnd.Next(0, 10000)}" );
    }

    private void Remove()
    {
        _steps.Remove(_steps[^1]);
    }
}
